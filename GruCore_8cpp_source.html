122 void GruCore::forward(dtype2 *prevAct, dtype2 *act, dtype2 *params,
123  unsigned int *h_inputLengths, unsigned int *d_inputLengths, bool deriv,
124  dtype2 *priorAct, int s) {
125  dtype2* wz_x = params + wzxOffset;
126  dtype2* wz_h = params + wzhOffset;
127  dtype2* bz = params + bzOffset;
128  dtype2* wr_x = params + wrxOffset;
129  dtype2* wr_h = params + wrhOffset;
130  dtype2* br = params + brOffset;
131  dtype2* wh_x = params + whxOffset;
132  dtype2* wh_h = params + whhOffset;
133  dtype2* bh = params + bhOffset;
134  dtype2* brec = params + brecOffset;
135 
136  dtype2* x_t = prevAct;
137  dtype2* h_t = act;
138 
139  dtype2* z = zs[s];
140  dtype2* r = rs[s];
141  dtype2* h_candidate = h_candidates[s];
142 
143  dtype2* h_prev;
144  if (s == 0) {
145  size_t arraySize = batchSize * size * sizeof(dtype2);
146  checkCudaErrors(cudaMemset(recBias, 0, arraySize));
147  CudaFunc::addRowVecMat(recBias, brec, batchSize, size, batchSize);
148  h_prev = recBias;
149  } else {
150  h_prev = priorAct;
151  }
152 
153 
154  // Step 1: Compute reset gate (r) and update gate (z) activations
155  // r = sigmoid(Wr_x * x_t + Wr_h * h_prev + br)
156  // z = sigmoid(Wz_x * x_t + Wz_h * h_prev + bz)
157  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, batchSize, size, preSize, &one,
158  x_t, batchSize, wr_x, preSize, &zero, r, batchSize);
159  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, batchSize, size, size, &one,
160  h_prev, batchSize, wr_h, size, &one, r, batchSize);
161  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, batchSize, size, preSize, &one,
162  x_t, batchSize, wz_x, preSize, &zero, z, batchSize);
163  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, batchSize, size, size, &one,
164  h_prev, batchSize, wz_h, size, &one, z, batchSize);
165  CudaFunc::addRowVecMat(r, br, batchSize, size, batchSize);
166 
167  CudaFunc::addRowVecMat(z, bz, batchSize, size, batchSize);
168  CudaFunc::sigmoid(r, r, batchSize * size);
169  CudaFunc::sigmoid(z, z, batchSize * size);
170 
171  // mask for varying sentence lengths
172  CudaFunc::maskByLength(r, d_inputLengths, s, size, batchSize);
173  CudaFunc::maskByLength(z, d_inputLengths, s, size, batchSize);
174 
175  // Step 2: Compute candidate hidden state (h_candidate) activation
176  // h_candidate = tanh(Wh_x * x_t + Wh_h * (r * h_prev) + bh)
177  CudaFunc::multiply(r, h_prev, intermed1, batchSize * size);
178  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, batchSize, size, preSize, &one,
179  x_t, batchSize, wh_x, preSize, &zero, h_candidate, batchSize);
180  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, batchSize, size, size, &one,
181  intermed1, batchSize, wh_h, size, &one, h_candidate, batchSize);
182  CudaFunc::addRowVecMat(h_candidate, bh, batchSize, size, batchSize);
183  CudaFunc::tanh(h_candidate, h_candidate, batchSize * size);
184 
185  CudaFunc::maskByLength(h_candidate, d_inputLengths, s, size, batchSize);
186 
187  // Step 3: Update the hidden state (h_t)
188  // h_t = (1 - z) * h_prev + z * h_candidate
189  CudaFunc::multiply(h_candidate, z, h_t, batchSize * size);
190  CudaFunc::subtractFromOne(z, intermed2, batchSize * size);
191  CudaFunc::multiply(h_prev, intermed2, intermed1, batchSize * size);
192  CudaFunc::add(intermed1, h_t, h_t, batchSize * size);
193 
194  CudaFunc::maskByLength(h_t, d_inputLengths, s, size, batchSize);
195 }

197 void GruCore::calcGrad(dtype2 *prevAct, dtype2 *act, dtype2 *prevError, dtype2 *error,
198  dtype2 *grad, dtype2 *params, unsigned int *h_inputLengths,
199  unsigned int *d_inputLengths, dtype2 *priorAct, dtype2 *priorError, int s) {
200  dtype2* wz_x = params + wzxOffset;
201  dtype2* wz_h = params + wzhOffset;
202  dtype2* wr_x = params + wrxOffset;
203  dtype2* wr_h = params + wrhOffset;
204  dtype2* wh_x = params + whxOffset;
205  dtype2* wh_h = params + whhOffset;
206 
207  dtype2* dwz_x = grad + wzxOffset;
208  dtype2* dwz_h = grad + wzhOffset;
209  dtype2* dbz = grad + bzOffset;
210  dtype2* dwr_x = grad + wrxOffset;
211  dtype2* dwr_h = grad + wrhOffset;
212  dtype2* dbr = grad + brOffset;
213  dtype2* dwh_x = grad + whxOffset;
214  dtype2* dwh_h = grad + whhOffset;
215  dtype2* dbh = grad + bhOffset;
216  dtype2* dbrec = grad + brecOffset;
217 
218  dtype2* h_t = act;
219  dtype2* h_prev = (s > 0)?priorAct:recBias;
220  dtype2* x_t = prevAct;
221 
222  dtype2* z = zs[s];
223  dtype2* r = rs[s];
224  dtype2* h_candidate = h_candidates[s];
225 
226  // Calculate dh_candidate, dr, and dz
227  // Compute the gradients of the candidate hidden state (dh_candidate)
228  CudaFunc::multiply(error, z, dh_candidate, batchSize * size);
229 
230  // Compute the gradients of the reset gate (dr)
231  // Compute the pointwise product (1 - h_candidate^2)
232  CudaFunc::multiply(h_candidate, h_candidate, intermed1, size * batchSize);
233  CudaFunc::subtractFromOne(intermed1, intermed2, size * batchSize);
234 
235  // Compute Dh_candidate = (dh_candidate ⊙ one_minus_h_c_2)
236  CudaFunc::multiply(dh_candidate, intermed2, dh_candidate, batchSize * size);
237 
238  // Compute ((dh_candidate ⊙ one_minus_h_c_2) * Wh_h^T)
239  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, size, size, &one,
240  dh_candidate, batchSize, wh_h, size, &zero, intermed2, batchSize);
241 
242  // Compute dr = (((Dh_candidate ⊙ one_minus_h_c_2) * Wh_h^T) ⊙ h_prev)
243  CudaFunc::multiply(intermed2, h_prev, dr, batchSize * size);
244 
245  // Compute the gradients of the update gate (dz)
246  CudaFunc::subtract(h_candidate, h_prev, intermed2, batchSize * size); // intermed2 = h_candidate - h_prev
247  CudaFunc::multiply(error, intermed2, dz, batchSize * size); // dz = error ⊙ (h_candidate - h_prev)
248 
249  // multiply dr, dz, dh_candidate with the derivative of their activation function
250  //Dr = dr ⊙ r ⊙ (1-r)
251  CudaFunc::subtractFromOne(r, intermed1, batchSize * size);
252  CudaFunc::multiply(r, intermed1, intermed2, batchSize * size);
253  CudaFunc::multiply(intermed2, dr, dr, batchSize * size);
254 
255  //Dz = dz ⊙ z ⊙ (1-z)
256  CudaFunc::subtractFromOne(z, intermed1, batchSize * size);
257  CudaFunc::multiply(z, intermed1, intermed2, batchSize * size);
258  CudaFunc::multiply(intermed2, dz, dz, batchSize * size);
259 
260  // mask for varying sentence lengths
261  CudaFunc::maskByLength(dr, d_inputLengths, s, size, batchSize);
262  CudaFunc::maskByLength(dz, d_inputLengths, s, size, batchSize);
263  CudaFunc::maskByLength(dh_candidate, d_inputLengths, s, size, batchSize);
264 
265  // Compute gradients for Wz_x, Wr_x, Wh_x
266  CublasFunc::gemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, preSize, size, batchSize, &one,
267  x_t, batchSize, dz, batchSize, &one, dwz_x, preSize);
268  CublasFunc::gemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, preSize, size, batchSize, &one,
269  x_t, batchSize, dr, batchSize, &one, dwr_x, preSize);
270  CublasFunc::gemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, preSize, size, batchSize, &one,
271  x_t, batchSize, dh_candidate, batchSize, &one, dwh_x, preSize);
272 
273  // Compute gradients for Wz_h, Wr_h, Wh_h and biases bz, br, bh
274  // Compute the pointwise product r ⊙ h_prev
275  CudaFunc::multiply(r, h_prev, intermed1, size * batchSize);
276 
277  CublasFunc::gemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, size, size, batchSize, &one,
278  h_prev, batchSize, dz, batchSize, &one, dwz_h, size);
279  CublasFunc::gemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, size, size, batchSize, &one,
280  h_prev, batchSize, dr, batchSize, &one, dwr_h, size);
281  CublasFunc::gemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, size, size, batchSize, &one,
282  intermed1, batchSize, dh_candidate, batchSize, &one, dwh_h, size);
283 
284  CudaFunc::sum_cols_reduce4(dz, dbz, batchSize, size);
285  CudaFunc::sum_cols_reduce4(dr, dbr, batchSize, size);
286  CudaFunc::sum_cols_reduce4(dh_candidate, dbh, batchSize, size);
287 
288  // Backpropagate the error to the previous layer
289  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, preSize, size, &one,
290  dz, batchSize, wz_x, preSize, &one, prevError, batchSize);
291  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, preSize, size, &one,
292  dr, batchSize, wr_x, preSize, &one, prevError, batchSize);
293  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, preSize, size, &one,
294  dh_candidate, batchSize, wh_x, preSize, &one, prevError, batchSize);
295 
296  if (s == 0) {
297  size_t arraySize = batchSize * size * sizeof(dtype2);
298  checkCudaErrors(cudaMemset(intermed3, 0, arraySize));
299  priorError = intermed3;
300  }
301 
302  // Backpropagate the error to the prior time step
303  // First term: error ⊙ (1 - z)
304  CudaFunc::subtractFromOne(z, intermed1, batchSize * size); // Compute (1 - z)
305  CudaFunc::multiply(error, intermed1, intermed2, batchSize * size); // Compute error ⊙ (1 - z)
306 
307  // Add the first term to priorError
308  CudaFunc::add(intermed2, priorError, priorError, batchSize * size);
309 
310  // Second term: Dz * Wz_h^T
311  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, size, size, &one,
312  dz, batchSize, wz_h, size, &one, priorError, batchSize); // Compute (z ⊙ (1 - z)) * Wz_h^T
313 
314  // Third term: (Dhc * Wh_h^T) ⊙ r
315  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, size, size, &one,
316  dh_candidate, batchSize, wh_h, size, &zero, intermed1, batchSize);
317  CudaFunc::multiply(intermed1, r, intermed2, batchSize * size);
318  CudaFunc::add(intermed2, priorError, priorError, batchSize * size);
319 
320  // Fourth term: Dr * Wr_h^T
321  CublasFunc::gemm(handle, CUBLAS_OP_N, CUBLAS_OP_T, batchSize, size, size, &one,
322  dr, batchSize, wr_h, size, &one, priorError, batchSize);
323 
324  if (s == 0) {
325  CudaFunc::sum_cols_reduce4(priorError, dbrec, batchSize, size);
326  }
327 }
